\begin{battle}{Incursion}
\battleinfo{rout}{4-5}{\robin, \frederick}
\prep{
	\inventory{
		\robinf
		\begin{itemize}
			\use{Second Seal}, reclass to Pegasus Rider
			\item Take Bronze Lance, Javelin
		\end{itemize}
		\frederickf \ ($\rightarrow$)
		\begin{itemize}
			\item Stash all weapons
		\end{itemize}
	}
	\units{
		\remove{\ricken \ ($\downarrow$)}
	}
}
\begin{enumerate}
\turn{
	\pair{\frederick}{\robin}
	\robinf
	\movewait{3R, 1U}
	\turnend
}
\turn{
	\robinf
	\moveattack{2U, 5R}{Archer}{1R}
	\auto
}
\turn{
	\item If you are low on HP (below half), can heal. Always end battle on turn 5 at the latest, to avoid reinforcements. Sometime the archer doesn't move to the correct spot.
	\item \textit{If you don't need to heal:}
	\begin{itemize}
	\robinf
	\moveattack{3R}{Archer}{1R}
	\end{itemize}
	\textit{If you do need to heal:}
	\begin{itemize}
	\robinf
	\moveattack{2R, 1D}{Archer}{1U}
	\end{itemize}
	\item \textit{If the \enemy{Archer} moved to 1R of \robin:}
	\begin{itemize}
	\robinf
	\moveattack{1R, 1D}{Archer}{1U}
	\end{itemize}
}
\turn{
	\item \textit{If you didn't need to heal:}
	\begin{itemize}
		\auto
	\end{itemize}
	\item \textit{If you do need to heal}
	\begin{itemize}
	\robinf
	\moveuse{2R}{Elixir/Concoction}
	\end{itemize}
	\item \textit{If the \enemy{Archer} had moved to 1R of \robin:}
	\begin{itemize}
	\robinf
	\moveuse{4R}{Elixir/Concoction if needed}
	\end{itemize}
}
\turn{
	\robinf
	\begin{itemize}
		\item Attack the \enemy{Bandit with the Hand Axe}
	\end{itemize}
	\auto\ if needed.
	\turnend
}
\end{enumerate}
\end{battle}
