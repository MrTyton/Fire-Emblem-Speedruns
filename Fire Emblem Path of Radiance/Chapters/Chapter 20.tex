\begin{battle}{Defending Talrega}
\battleinfo{Arive in 15 turns}{2}{\ike, \marcia, \titania, \tanith, \reyson, \calill}
\base{
    \item Info: Woman (\calill)
}
\prep{
    \units{
        \item Remove everyone below \reyson
        \item Select \calill
    }
    \map{
        \item Move \reyson\ 4U
        \item Move \calill\ 2U
        \item Move \tanith\ 1U, 1R
        \item Move \marcia\ 3U, 1R
        \item Move \titania\ 3U, 1L
    }
}
\textit{\underline{Note}: You may want to safety save because of Tanith’s small chances of dying.}
\vspace{3mm}
\begin{enumerate}
\turn{
    \ikef
    \movewait{1R}
    \tanithf
    \begin{itemize}
        \move{1R}
        \item Trade with \marcia: Short Spear for Javelin \#3 and Killer Lance
        \unequip
    \end{itemize}
    \canto{8R}
    \Xoff{\marcia}
    \begin{itemize}
        \unequip
        \item Rescue \ike
    \end{itemize}
    \canto{4R, 4U}
    \Xmove{1}{\titania}
    \moveattack{5R, 4U}{Halberdier}{1U or 1R}
    \Xmove{1}{\reyson}
    \begin{itemize}
        \move{4R, 1U}
        \item Chant \marcia
    \end{itemize}
    \Xmove{1}{\marcia}
    \begin{itemize}
        \move{11R}
        \drop{\ike\ right}
    \end{itemize}
    \Xmove{1}{\calill}
    \moveattack{3R, 3U}{Halberdier}{2R or 1U, 1R with \textbf{Elthunder}}
}
\turn{
    \Xmove{2}{\titania}
    \begin{itemize}
        \move{1R, 2U}
        \item Visit Village for Smite (send to Convoy)
    \end{itemize}
    \Xmove{1}{\tanith}
    \begin{itemize}
        \move{4D, 3R}
        \item Visit Village for Rescue
    \end{itemize}
    \item (1X 1U) \calill:
    \moveattack{5R}{Shiharam}{10R \begin{itemize}
        \item \textit{If she misses, Chant her with \reyson}
    \end{itemize}}
    \Xoff{\ike}
    \moveattack{2R}{Shiharam}{1U}
    \Xmove{1}{\marcia}
    \begin{itemize}
        \item Arrive
    \end{itemize}
}
\end{enumerate}
\vspace{2mm}
\begin{itemize}
    \item \textit{There’s a small chance that a \enemy{Wyvern} will block the bottom village. You can move \calill\ closer after attacking the boss by using \reyson\ 's Chant, then snipe the \enemy{Wyvern} with Meteor (she 2X)}
\end{itemize}
\end{battle}
